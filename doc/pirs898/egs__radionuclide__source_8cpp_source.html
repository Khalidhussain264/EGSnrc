42 EGS_RadionuclideSource::EGS_RadionuclideSource(EGS_Input *input,
43  EGS_ObjectFactory *f) : EGS_BaseSource(input,f), shape(0),
44  geom(0), regions(0), nrs(0), min_theta(0), max_theta(M_PI),
45  min_phi(0), max_phi(2*M_PI), gc(IncludeAll), q_allowed(0), decays(0),
46  activity(0) {
47 
48  int err;
49  vector<int> tmp_q;
50  err = input->getInput("charge", tmp_q);
51  if (!err) {
52  if (std::find(q_allowed.begin(), q_allowed.end(), -1) != q_allowed.end()
53  && std::find(q_allowed.begin(), q_allowed.end(), 0) != q_allowed.end()
54  && std::find(q_allowed.begin(), q_allowed.end(), 1) != q_allowed.end()
55  && std::find(q_allowed.begin(), q_allowed.end(), 2) != q_allowed.end()
56  ) {
57  q_allowAll = true;
58  }
59  else {
60  q_allowAll = false;
61  }
62  q_allowed = tmp_q;
63  }
64  else {
65  q_allowAll = true;
66  q_allowed.push_back(-1);
67  q_allowed.push_back(1);
68  q_allowed.push_back(0);
69  q_allowed.push_back(2);
70  }
71 
72  // Create the decay spectra
73  count = 0;
74  Emax = 0;
75  unsigned int i = 0;
76  EGS_Float spectrumWeightTotal = 0;
77  disintegrationOccurred = true;
78  ishower = -1;
79  time = 0;
80  while (input->getInputItem("spectrum")) {
81 
82  egsInformation("**********************************************\n");
83 
84  decays.push_back(EGS_BaseSpectrum::createSpectrum(input));
85 
86  // If spectrum creation failed skip to the next spectrum block
87  if (!decays[i]) {
88  decays.pop_back();
89  continue;
90  }
91 
92  EGS_Float spectrumMaxE = decays[i]->maxEnergy();
93  if (spectrumMaxE > Emax) {
94  Emax = spectrumMaxE;
95  }
96 
97  spectrumWeightTotal += decays[i]->getSpectrumWeight();
98 
99  ++i;
100  }
101  if (decays.size() < 1) {
102  egsWarning("EGS_RadionuclideSource: Error: No spectrum was defined\n");
103  return;
104  }
105 
106  // Normalize the spectrum weights
107  for (i=0; i<decays.size(); ++i) {
108  decays[i]->setSpectrumWeight(
109  decays[i]->getSpectrumWeight() / spectrumWeightTotal);
110 
111  if (i > 0) {
112  decays[i]->setSpectrumWeight(
113  decays[i]->getSpectrumWeight() +
114  decays[i-1]->getSpectrumWeight());
115  }
116  }
117 
118  // Get the activity
119  EGS_Float tmp_A;
120  err = input->getInput("activity", tmp_A);
121  if (!err) {
122  activity = tmp_A;
123  }
124  else {
125  activity = 1;
126  }
127 
128  egsInformation("EGS_RadionuclideSource: Activity [disintegrations/s]: %e\n",
129  activity);
130 
131  // Get the active application
132  app = EGS_Application::activeApplication();
133 
134  // Create the shape for source emissions
135  vector<EGS_Float> pos;
136  EGS_Input *ishape = input->takeInputItem("shape");
137  if (ishape) {
138  shape = EGS_BaseShape::createShape(ishape);
139  delete ishape;
140  }
141  if (!shape) {
142  string sname;
143  err = input->getInput("shape name",sname);
144  if (err)
145  egsWarning("EGS_RadionuclideSource: missing/wrong inline shape "
146  "definition and missing wrong 'shape name' input\n");
147  else {
148  shape = EGS_BaseShape::getShape(sname);
149  if (!shape) egsWarning("EGS_RadionuclideSource: a shape named %s"
150  " does not exist\n");
151  }
152  }
153  string geom_name;
154  err = input->getInput("geometry",geom_name);
155  if (!err) {
156  geom = EGS_BaseGeometry::getGeometry(geom_name);
157  if (!geom) egsWarning("EGS_RadionuclideSource: no geometry named %s\n",
158  geom_name.c_str());
159  else {
160  vector<string> reg_options;
161  reg_options.push_back("IncludeAll");
162  reg_options.push_back("ExcludeAll");
163  reg_options.push_back("IncludeSelected");
164  reg_options.push_back("ExcludeSelected");
165  gc = (GeometryConfinement) input->getInput("region "
166  "selection",reg_options,0);
167  if (gc == IncludeSelected || gc == ExcludeSelected) {
168  vector<int> regs;
169  err = input->getInput("selected regions",regs);
170  if (err || regs.size() < 1) {
171  egsWarning("EGS_RadionuclideSource: region selection %d "
172  "used but no 'selected regions' input "
173  "found\n",gc);
174  gc = gc == IncludeSelected ? IncludeAll : ExcludeAll;
175  egsWarning(" using %d\n",gc);
176  }
177  nrs = regs.size();
178  regions = new int [nrs];
179  for (int j=0; j<nrs; j++) {
180  regions[j] = regs[j];
181  }
182  }
183  }
184  }
185  EGS_Float tmp_theta;
186  err = input->getInput("min theta", tmp_theta);
187  if (!err) {
188  min_theta = tmp_theta/180.0*M_PI;
189  }
190 
191  err = input->getInput("max theta", tmp_theta);
192  if (!err) {
193  max_theta = tmp_theta/180.0*M_PI;
194  }
195 
196  err = input->getInput("min phi", tmp_theta);
197  if (!err) {
198  min_phi = tmp_theta/180.0*M_PI;
199  }
200 
201  err = input->getInput("max phi", tmp_theta);
202  if (!err) {
203  max_phi = tmp_theta/180.0*M_PI;
204  }
205 
206  buf_1 = cos(min_theta);
207  buf_2 = cos(max_theta);
208 
209  setUp();

212 EGS_I64 EGS_RadionuclideSource::getNextParticle(EGS_RandomGenerator *rndm, int
213  &q, int &latch, EGS_Float &E, EGS_Float &wt, EGS_Vector &x, EGS_Vector
214  &u) {
215 
216  unsigned int i = 0;
217  if (decays.size() > 1 && disintegrationOccurred) {
218  // Sample a uniform random number
219  EGS_Float uRand = rndm->getUniform();
220 
221  // Sample which spectrum to use
222  for (i=0; i<decays.size(); ++i) {
223  if (uRand < decays[i]->getSpectrumWeight()) {
224  break;
225  }
226  }
227  }
228 
229  EGS_I64 ishowerOld = decays[i]->getShowerIndex();
230 
231  for (EGS_I64 j=0; j<=1e6; ++j) {
232 
233  E = decays[i]->sampleEnergy(rndm);
234 
235  // Skip zero energy particles
236  if (E < epsilon) {
237  continue;
238  }
239 
240  q = decays[i]->getCharge();
241 
242  // Check if the charge is allowed
243  // If so, break out of the loop and keep the particle
244  // Otherwise the loop will continue generating particles until
245  // one matches the q_allowed criteria
246  if (q_allowAll || std::find(q_allowed.begin(), q_allowed.end(), q) != q_allowed.end()) {
247  break;
248  }
249 
250  if (j == 1e6) {
251  egsWarning("EGS_RadionuclideSource::getNextParticle: Error: Could not generate a particle after 1e6 tries. Spectrum will be wrong.\n");
252  E = 0;
253  }
254  }
255 
256  EGS_I64 ishowerNew = decays[i]->getShowerIndex();
257  if (ishowerNew > ishowerOld) {
258  disintegrationOccurred = true;
259 
260  time += -log(1.-rndm->getUniform()) / activity * (ishowerNew - ishowerOld);
261  ishower += (ishowerNew - ishowerOld);
262  }
263  else {
264  disintegrationOccurred = false;
265 
266  time += decays[i]->getTime();
267  }
268 
269  getPositionDirection(rndm,x,u,wt);
270  latch = 0;
271 
272  if (disintegrationOccurred) {
273  xOfDisintegration = x;
274  }
275  else {
276  x = xOfDisintegration;
277  }
278 
279  // Now that we have the position of the particle, we can
280  // find the region and deposit any sub-threshold contributions locally
281  // This includes edep from relaxations, and alpha particle energy
282  EGS_Float edep = decays[i]->getEdep();
283  if (edep > 0) {
284  app->setEdep(edep);
285  int ireg = app->isWhere(x);
286  app->userScoring(3, ireg);
287  }
288 
289  return ++count;
290 }

292 void EGS_RadionuclideSource::getPositionDirection(EGS_RandomGenerator *rndm,
293  EGS_Vector &x, EGS_Vector &u, EGS_Float &wt) {
294 
295  bool ok = true;
296  do {
297  x = shape->getRandomPoint(rndm);
298  if (geom) {
299  if (gc == IncludeAll) {
300  ok = geom->isInside(x);
301  }
302  else if (gc == ExcludeAll) {
303  ok = !geom->isInside(x);
304  }
305  else if (gc == IncludeSelected) {
306  ok = false;
307  int ireg = geom->isWhere(x);
308  for (int j=0; j<nrs; ++j) {
309  if (ireg == regions[j]) {
310  ok = true;
311  break;
312  }
313  }
314  }
315  else {
316  ok = true;
317  int ireg = geom->isWhere(x);
318  for (int j=0; j<nrs; ++j) {
319  if (ireg == regions[j]) {
320  ok = false;
321  break;
322  }
323  }
324  }
325  }
326  }
327  while (!ok);
328  u.z = buf_1 - rndm->getUniform()*(buf_1 - buf_2);
329  EGS_Float sinz = 1-u.z*u.z;
330  if (sinz > epsilon) {
331  sinz = sqrt(sinz);
332  EGS_Float cphi, sphi;
333  EGS_Float phi = min_phi +(max_phi - min_phi)*rndm->getUniform();
334  cphi = cos(phi);
335  sphi = sin(phi);
336  u.x = sinz*cphi;
337  u.y = sinz*sphi;
338  }
339  else {
340  u.x = 0;
341  u.y = 0;
342  }
343  wt = 1;
344 }
